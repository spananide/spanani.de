    <script>
        // --- Calculator 1: Steam Investment Logic ---
        const buyPriceInput = document.getElementById('buyPrice');
        const targetProfitInput = document.getElementById('targetProfit');
        
        const outBreakEven = document.getElementById('outBreakEven');
        const outTargetSell = document.getElementById('outTargetSell');
        const outRustFee = document.getElementById('outRustFee');
        const outSteamFee = document.getElementById('outSteamFee');

        function calculateSteamInvestment() {
            let buyPrice = parseFloat(buyPriceInput.value) || 0;
            let targetProfit = parseFloat(targetProfitInput.value) || 0;

            // Break Even = Buy Price * 1.15
            let breakEven = buyPrice * 1.15;
            
            // Target Net = What we want to end up in our wallet
            let targetNet = buyPrice + targetProfit;
            
            // Target Gross = What the buyer pays (Target Net * 1.15)
            let targetGross = targetNet * 1.15;

            // Fees are based on the Net amount
            let rustFee = targetNet * 0.10;
            let steamFee = targetNet * 0.05;

            outBreakEven.innerText = "$" + breakEven.toFixed(2);
            outTargetSell.innerText = "$" + targetGross.toFixed(2);
            outRustFee.innerText = "$" + rustFee.toFixed(2);
            outSteamFee.innerText = "$" + steamFee.toFixed(2);
        }

        buyPriceInput.addEventListener('input', calculateSteamInvestment);
        targetProfitInput.addEventListener('input', calculateSteamInvestment);

        // --- Calculator 2: Arbitrage Logic ---
        const cashPriceInput = document.getElementById('cashPrice');
        const steamPriceInput = document.getElementById('steamPrice');

        const outSteamWallet = document.getElementById('outSteamWallet');
        const outArbitrage = document.getElementById('outArbitrage');
        const arbitrageBox = document.getElementById('arbitrageBox');
        const arbitrageVerdict = document.getElementById('arbitrageVerdict');

        function calculateArbitrage() {
            let cashPrice = parseFloat(cashPriceInput.value) || 0;
            let steamListing = parseFloat(steamPriceInput.value) || 0;

            // Steam divides the gross listing by 1.15 to determine the seller's net wallet funds
            let walletReceived = steamListing / 1.15;
            let netGain = walletReceived - cashPrice;
            let roiPercentage = cashPrice > 0 ? (netGain / cashPrice) * 100 : 0;

            outSteamWallet.innerText = "$" + walletReceived.toFixed(2);
            
            if (netGain >= 0) {
                outArbitrage.innerText = "+$" + netGain.toFixed(2);
                outArbitrage.classList.remove('text-red-400');
                outArbitrage.classList.add('text-green-400');
                arbitrageBox.classList.add('bg-green-500/10', 'border-green-500/30');
                arbitrageBox.classList.remove('bg-red-500/10', 'border-red-500/30', 'bg-black/30');
                arbitrageVerdict.innerText = `This flip yields a +${roiPercentage.toFixed(2)}% increase in Steam Wallet funds.`;
            } else {
                outArbitrage.innerText = "-$" + Math.abs(netGain).toFixed(2);
                outArbitrage.classList.remove('text-green-400');
                outArbitrage.classList.add('text-red-400');
                arbitrageBox.classList.add('bg-red-500/10', 'border-red-500/30');
                arbitrageBox.classList.remove('bg-green-500/10', 'border-green-500/30', 'bg-black/30');
                arbitrageVerdict.innerText = `You will lose ${Math.abs(roiPercentage).toFixed(2)}% of your value making this flip.`;
            }
        }

        cashPriceInput.addEventListener('input', calculateArbitrage);
        steamPriceInput.addEventListener('input', calculateArbitrage);

        // Run once on load to populate initial math
        calculateSteamInvestment();
        calculateArbitrage();
    </script>
